script.js 
// llm_vis base shell (no charts yet) — D3 v7

const DATE_PARSE = d3.timeParse("%Y-%m-%d");

// Wire up once the DOM is ready
document.addEventListener("DOMContentLoaded", () => {
  // Controls
  d3.select("#dataset").on("change", changeData);
  d3.select("#metric").on("change", changeData);

  // Create empty SVG scaffolds for each container (optional, handy later)
  ["Histogram-div","Linechart-div","StackedArea-div","Scatterplot-div"].forEach(id => {
    const svg = d3.select("#" + id).append("svg")
      .attr("viewBox", "0 0 500 260"); // matches CSS height
    // Keep axes groups ready for future steps
    svg.append("g").attr("class", "plot").attr("transform", "translate(50,20)");
    svg.append("g").attr("class", "x-axis").attr("transform", "translate(50,230)");
    svg.append("g").attr("class", "y-axis").attr("transform", "translate(50,20)");
  });

  // Initial load
  changeData();
});

// Loads the selected CSV and logs parsed rows; charts will be added in later prompts
function changeData() {
  const file = d3.select("#dataset").property("value");       // e.g., "route_b.csv"
  const metric = d3.select("#metric").property("value");      // "ridership" or "on_time_pct"

  console.log("Loading:", `data/${file}`, "metric:", metric);

  d3.csv(`data/${file}`).then(raw => {
    // Parse to the known schema; keep as strings where not needed yet
    const rows = raw.map(d => ({
      date: DATE_PARSE(d.date),
      group: d.group,
      ridership: +d.ridership,
      on_time_pct: +d.on_time_pct,
      day_type: d.day_type
    })).filter(d => d.date);

    rows.sort((a, b) => a.date - b.date);

    // For this base shell we just log; next steps will draw each chart.
    console.log(`Loaded ${rows.length} rows from ${file}. Example row:`, rows[0]);
  }).catch(err => {
    console.error("Failed to load CSV:", err);
    alert("Error loading CSV. Check that data/ exists and filenames match the dropdown.");
  });
}
